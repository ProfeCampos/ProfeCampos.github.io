const int pinSensor = A0;
const int pinVentilador = 9;
const int pinFoco = 7;
const float UMBRAL_VENTILADOR = 28.0;
const float UMBRAL_PELIGRO = 35.0;
LiquidCrystal_I2C lcd(0x27, 16, 2);

void setup() {
  pinMode(pinVentilador, OUTPUT);
  pinMode(pinFoco, OUTPUT);
  lcd.init();
  lcd.backlight();
}
void loop() {
  float temp = ((analogRead(pinSensor)/1023.0)*5.0)*100;
  lcd.clear();
  lcd.print("Temp: "); lcd.print(temp); lcd.print(" C");
  
  if (temp > UMBRAL_PELIGRO) {
    lcd.setCursor(0,1); lcd.print("PELIGRO");
    digitalWrite(pinVentilador, HIGH);
    digitalWrite(pinFoco, HIGH);
  } else if (temp > UMBRAL_VENTILADOR) {
    lcd.setCursor(0,1); lcd.print("Enfriando...");
    digitalWrite(pinVentilador, HIGH);
    digitalWrite(pinFoco, LOW);
  } else {
    lcd.setCursor(0,1); lcd.print("Estado: Normal");
    digitalWrite(pinVentilador, LOW);
    digitalWrite(pinFoco, LOW);
  }
  delay(1000);
}
