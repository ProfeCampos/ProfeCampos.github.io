void loop() {
  valorLuz = analogRead(pinLdr);
  hayMovimiento = digitalRead(pinPir);
  
  if (hayMovimiento == true && valorLuz < umbralOscuridad) {
    digitalWrite(pinLed, HIGH);
    delay(5000); // Mantiene la luz encendida 5 segundos
  } else {
    digitalWrite(pinLed, LOW);
  }
  delay(300);
}
